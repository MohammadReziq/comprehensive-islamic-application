================================================================================
  04 — التأثير المتبادل بين الأدوار
  تطبيق "صلاتي حياتي" — الخطة النهائية
================================================================================

الفهرس:
  1. أفعال السوبر أدمن وتأثيرها
  2. أفعال الإمام وتأثيرها
  3. أفعال المشرف وتأثيرها
  4. أفعال ولي الأمر وتأثيرها
  5. ملخص التأثيرات المتقاطعة


================================================================================
  1. أفعال السوبر أدمن وتأثيرها على الآخرين
================================================================================

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  1.1 أنشأ مسابقة عامة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  قبل التفعيل: لا أحد يتأثر (المسابقة مخفية).

  بعد التفعيل:
  ┌──────────────┬──────────────────────────────────────────┐
  │ الدور        │ الأثر                                    │
  ├──────────────┼──────────────────────────────────────────┤
  │ الإمام       │ يرى بانر "مسابقة عامة" في قسم المسابقات │
  │              │ يرى اسمها + مدتها + نقاطها               │
  │              │ يرى ترتيب مسجده + الترتيب العام           │
  │              │ لا يحتاج يعمل شيء إضافي                  │
  ├──────────────┼──────────────────────────────────────────┤
  │ المشرف       │ يرى بانر "مسابقة عامة جارية" فوق الترتيب│
  │              │ يستمر بتسجيل الحضور كالعادة               │
  │              │ لا يحتاج يعمل شيء إضافي                  │
  ├──────────────┼──────────────────────────────────────────┤
  │ ولي الأمر    │ يرى بانر ذهبي على الرئيسية               │
  │              │ يرى ترتيب ابنه في المسابقة العامة         │
  │              │ يرى ترتيب ابنه في المحلية (لو موجودة)     │
  └──────────────┴──────────────────────────────────────────┘

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  1.2 غيّر نقاط صلاة أثناء مسابقة عامة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  ⚠ في MVP: لا نسمح بالتغيير أثناء المسابقة.

  مستقبلاً (لو سُمح):
  ┌──────────────┬──────────────────────────────────────────┐
  │ الدور        │ الأثر                                    │
  ├──────────────┼──────────────────────────────────────────┤
  │ الإمام       │ يرى إشعار "تم تعديل نقاط المسابقة"      │
  │              │ النقاط الجديدة تظهر في تفاصيل المسابقة   │
  ├──────────────┼──────────────────────────────────────────┤
  │ المشرف       │ لا شيء يتغير (يسجل حضور كالعادة)        │
  ├──────────────┼──────────────────────────────────────────┤
  │ ولي الأمر    │ ترتيب ابنه قد يتغير (لأن النقاط تغيرت) │
  │              │ يرى إشعار "تم تعديل نقاط المسابقة"      │
  ├──────────────┼──────────────────────────────────────────┤
  │ الحضور السابق│ لا يتأثر (النقاط القديمة ثابتة)         │
  │ الحضور القادم│ يُحسب بالنقاط الجديدة                   │
  └──────────────┴──────────────────────────────────────────┘

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  1.3 علّق مسجد أثناء مسابقة عامة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  ┌──────────────┬──────────────────────────────────────────┐
  │ العنصر       │ الأثر                                    │
  ├──────────────┼──────────────────────────────────────────┤
  │ نقاط الأطفال │ تبقى كما هي (لا تُحذف)                  │
  │ الترتيب      │ الأطفال يظهرون بـ "(مسجد موقوف)"        │
  │ حضور جديد    │ لا يُسجَّل (المسجد موقوف)                │
  │ الإمام       │ يرى رسالة "مسجدك موقوف"                 │
  │ المشرفون     │ لا يستطيعون التسجيل                      │
  │ أولياء الأمور│ يرون حالة الحضور القديمة فقط              │
  │ بعد إعادة    │ كل شيء يعود طبيعي — الحضور يُسجَّل      │
  │ التفعيل      │ والنقاط تُحسب                            │
  └──────────────┴──────────────────────────────────────────┘

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  1.4 أضاف مسجد لمسابقة قائمة (بعد البدء)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  ┌──────────────┬──────────────────────────────────────────┐
  │ العنصر       │ الأثر                                    │
  ├──────────────┼──────────────────────────────────────────┤
  │ المسجد الجديد│ يبدأ من 0 نقاط                          │
  │ أطفاله       │ يظهرون في الترتيب العام (آخر المراكز)    │
  │ الإمام       │ يرى "مسابقة عامة" فجأة في لوحته         │
  │ المشرف       │ يرى بانر "مسابقة عامة جديدة"            │
  │ المساجد      │ لا تتأثر — ترتيبها يبقى                  │
  │ الأخرى       │                                          │
  └──────────────┴──────────────────────────────────────────┘

  ⚠ تنبيه للسوبر أدمن: "المسجد سيبدأ متأخراً بـ X أيام."

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  1.5 حظر إمام أثناء مسابقة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  ┌──────────────┬──────────────────────────────────────────┐
  │ العنصر       │ الأثر                                    │
  ├──────────────┼──────────────────────────────────────────┤
  │ المسجد       │ يصبح "يتيم" (بلا إمام)                   │
  │ المشرفون     │ يستمرون بالتسجيل (لا يتأثرون)           │
  │ المسابقة     │ تستمر (لأنها مرتبطة بالمسجد لا بالإمام) │
  │ المحلية      │ المسابقة المحلية لا يقدر أحد يديرها     │
  │              │ (تبقى نشطة حتى end_date أو السوبر أدمن) │
  │ السوبر أدمن  │ يعيّن إمام بديل في أقرب وقت             │
  └──────────────┴──────────────────────────────────────────┘


================================================================================
  2. أفعال الإمام وتأثيرها على الآخرين
================================================================================

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  2.1 أنشأ مسابقة محلية
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  ┌──────────────┬──────────────────────────────────────────┐
  │ الدور        │ الأثر                                    │
  ├──────────────┼──────────────────────────────────────────┤
  │ السوبر أدمن  │ لا يتأثر (لا يرى المسابقات المحلية)     │
  │ المشرف       │ يرى المسابقة + الترتيب (قراءة فقط)      │
  │ ولي الأمر    │ يرى بانر "مسابقة في مسجد ابنك"          │
  └──────────────┴──────────────────────────────────────────┘

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  2.2 أزال مشرف أثناء مسابقة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  ┌──────────────┬──────────────────────────────────────────┐
  │ العنصر       │ الأثر                                    │
  ├──────────────┼──────────────────────────────────────────┤
  │ الحضور الذي  │ يبقى كما هو (بيانات حقيقية — لا تُحذف) │
  │ سجّله        │                                          │
  │ النقاط       │ لا تتأثر                                 │
  │ الترتيب      │ لا يتأثر                                 │
  │ المشرف نفسه  │ يفقد الوصول فوراً                        │
  │ المسجد       │ أصبح بمشرف أقل — تحذير لو بقي بلا مشرفين│
  └──────────────┴──────────────────────────────────────────┘

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  2.3 عدّل إعدادات المسجد أثناء مسابقة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  ■ غيّر prayer_config (نقاط الصلوات):
    → الحضور القادم يُحسب بالنقاط الجديدة
    → الحضور السابق لا يتأثر (لأنه مُخزّن في attendance.points_earned)
    → الترتيب قد يتغير تدريجياً

  ■ غيّر كود الدعوة:
    → لا تأثير على المسابقة
    → الكود القديم يتوقف عن العمل

  ■ غيّر اسم/موقع المسجد:
    → لا تأثير على المسابقة

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  2.4 أوقف المسابقة المحلية يدوياً
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  ┌──────────────┬──────────────────────────────────────────┐
  │ الدور        │ الأثر                                    │
  ├──────────────┼──────────────────────────────────────────┤
  │ المشرف       │ يرى "المسابقة انتهت" + النتائج النهائية │
  │ ولي الأمر    │ يرى "المسابقة انتهت" + مركز ابنه        │
  │ المسابقة     │ تتحول لـ "منتهية" في الأرشيف            │
  │ العامة       │                                          │
  │ الحضور       │ يستمر بالتسجيل ← لكن لا يُربط بمسابقة  │
  │              │ (competition_id = NULL)                   │
  └──────────────┴──────────────────────────────────────────┘


================================================================================
  3. أفعال المشرف وتأثيرها على الآخرين
================================================================================

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  3.1 سجّل حضور خاطئ
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  أثر فوري:
    → الطفل يكسب نقاط (Trigger 020)
    → الترتيب يتغير
    → ولي الأمر يرى "حضر" (Realtime)

  التصحيح:
    ■ المشرف: يلغي الحضور (خلال 24 ساعة) → Trigger يعيد الحساب
    ■ الإمام: يلغي الحضور (بلا حدود) → Trigger يعيد الحساب
    ■ ولي الأمر: يلاحظ ← لا يحتاج يعمل شيء (سيُلغى)

  النقاط والترتيب: يعودان تلقائياً بعد الإلغاء.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  3.2 لم يسجل حضور ليوم كامل (غائب)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  ┌──────────────┬──────────────────────────────────────────┐
  │ العنصر       │ الأثر                                    │
  ├──────────────┼──────────────────────────────────────────┤
  │ الأطفال      │ لا يكسبون نقاط لهذا اليوم                │
  │ السلاسل      │ تنقطع (current_streak يُعاد لـ 0)        │
  │ الترتيب      │ يتأثر (فقدوا يوم كامل من النقاط)         │
  │ ولي الأمر    │ يرى "لم يُسجَّل حضور" ← يتساءل          │
  │              │ يمكنه إرسال طلب تصحيح لليوم التالي       │
  │ الإمام       │ يرى في Audit: "يوم X لم يُسجَّل حضور"   │
  │              │ يتواصل مع المشرف                         │
  └──────────────┴──────────────────────────────────────────┘

  ⚠ هذا أخطر سيناريو — لأن الأطفال حضروا فعلياً لكن لم يُسجَّل!
  الحل: الإمام يسجل بأثر رجعي (override بلا حدود زمنية).

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  3.3 المشرف أُزيل من المسجد
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  → يفقد الوصول فوراً (لا يقدر يسجل دخول كمشرف)
  → كل الحضور الذي سجّله يبقى (مرتبط بـ recorded_by)
  → لو المسجد بلا مشرفين → تحذير الإمام
  → لا تأثير على المسابقة أو النقاط


================================================================================
  4. أفعال ولي الأمر وتأثيرها
================================================================================

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  4.1 أرسل طلب تصحيح أثناء المسابقة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  التأثير الفوري: لا شيء — الطلب "معلق" (pending).

  عند القبول:
  ┌──────────────┬──────────────────────────────────────────┐
  │ العنصر       │ الأثر                                    │
  ├──────────────┼──────────────────────────────────────────┤
  │ الحضور       │ يُضاف سجل attendance جديد               │
  │ النقاط       │ Trigger يحسب النقاط ← total_points يزيد  │
  │ competition_id│ Trigger 045 يعيّنه تلقائياً              │
  │ الترتيب      │ يتغير فوراً (لأن النقاط زادت)            │
  │ السلسلة      │ تُعاد حسابها (قد تعود لو كان يوم مفقود) │
  │ ولي الأمر    │ يرى "تم قبول طلبك" + تحديث الحضور       │
  └──────────────┴──────────────────────────────────────────┘

  عند الرفض: لا تأثير (الحضور لا يتغير).

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  4.2 فك ربط ابنه من المسجد أثناء المسابقة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  ┌──────────────┬──────────────────────────────────────────┐
  │ العنصر       │ الأثر                                    │
  ├──────────────┼──────────────────────────────────────────┤
  │ الحضور السابق│ يبقى في DB (لا يُحذف)                   │
  │ النقاط       │ تبقى في الترتيب                          │
  │ الترتيب      │ الطفل يظهر بـ "(منسحب)" لكن بنقاطه      │
  │ حضور جديد    │ لا يُسجَّل (لأنه لم يعد مربوطاً)        │
  │ المسابقة     │ لا تتأثر                                 │
  │ العامة       │ نفس الأثر                                │
  └──────────────┴──────────────────────────────────────────┘

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  4.3 أضاف ابن جديد أثناء المسابقة
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  ┌──────────────┬──────────────────────────────────────────┐
  │ العنصر       │ الأثر                                    │
  ├──────────────┼──────────────────────────────────────────┤
  │ الطفل الجديد │ يبدأ من 0 نقاط                          │
  │ الترتيب      │ يظهر في آخر القائمة                      │
  │ يكسب نقاط    │ من أول حضور يُسجَّل له                   │
  │ المسابقة     │ لا تتأثر (لا حاجة لتسجيله يدوياً)       │
  │ العامة       │ يظهر تلقائياً (لأن مسجده مشارك)          │
  └──────────────┴──────────────────────────────────────────┘


================================================================================
  5. ملخص التأثيرات المتقاطعة (جدول سريع)
================================================================================

  ┌─────────────────────────────┬─────┬─────┬─────┬─────┐
  │ الفعل                      │سوبر│إمام │مشرف │ أب  │
  ├─────────────────────────────┼─────┼─────┼─────┼─────┤
  │ إنشاء مسابقة عامة (سوبر)   │  —  │ يرى │ يرى │ يرى│
  │ إنشاء مسابقة محلية (إمام)  │  ❌ │  —  │ يرى │ يرى│
  │ تسجيل حضور (مشرف)          │  ❌ │ يرى │  —  │ يرى│
  │ طلب تصحيح (أب)             │  ❌ │ يرى │ يرى │  — │
  │ إلغاء حضور (إمام/مشرف)     │  ❌ │  —  │  —  │ يرى│
  │ تعليق مسجد (سوبر)          │  —  │يتأثر│يتأثر│يتأثر│
  │ إزالة مشرف (إمام)          │  ❌ │  —  │يتأثر│  ❌ │
  │ فك ربط طفل (أب)            │  ❌ │  ❌ │  ❌ │  — │
  └─────────────────────────────┴─────┴─────┴─────┴─────┘

  ❌ = لا يتأثر | يرى = يُبلَّغ | يتأثر = صلاحياته/بياناته تتغير
